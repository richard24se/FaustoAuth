#reiniciar el vagrant #SIEMPRE HACER ESTO LUEGO DEL PRIMER UP

#apagar el vagrant

vagrant provision

#conectar ssh
vagrant ssh {id}

#listar los vagrant #PARA UBICAR EL ID ÚNICO Y UTILIZAR EL SSH
vagrant global-status
